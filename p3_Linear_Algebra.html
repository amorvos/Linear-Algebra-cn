def calc_Norm(x, p=2, infty=False):
    if infty:
        return float(np.max(np.abs(x)))
    if p == 0:
        return float(np.sum(np.array(x) > 0))
    elif p > 0:
        return float(np.power(np.sum(np.power(x, p)), 1 / p))
    else:
        raise Exception(" >= 0")
